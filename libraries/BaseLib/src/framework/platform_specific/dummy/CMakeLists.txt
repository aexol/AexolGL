# Add sources for ios platform.

if(NOT ANDROID AND NOT IOS )
set(LOCAL_INCLUDES 
    )

ADD_CURRENT_LIST_DIR(SRC_FRAMEWORK_PLATFORM_SPECIFIC_DUMMY_INCLUDE)
set(SRC_FRAMEWORK_PLATFORM_SPECIFIC_DUMMY_INCLUDE
    ${SRC_FRAMEWORK_PLATFORM_SPECIFIC_DUMMY_INCLUDE}
    )

# Available dummy implementations.
# For each implementation there must exist a list
# of files needed to build it. The name must match
# {IMPLEMENTATION_NAME}_files.
set(_dummy_implementations
    ACCELEROMETER_IMPLEMENTATION
    ORIENTATION_IMPLEMENTATION
    CAMERA_IMPLEMENTATION
    DYNLOAD_IMPLEMENTATION
    OSUTILS_IMPLEMENTATION
    )
set(ACCELEROMETER_IMPLEMENTATION_files src/AccelerometerDummy.cpp)
set(ORIENTATION_IMPLEMENTATION_files src/OrientationImpl.cpp)
set(CAMERA_IMPLEMENTATION_files src/CameraDummy.cpp)
set(DYNLOAD_IMPLEMENTATION_files src/DynloadDummy.cpp)
set(OSUTILS_IMPLEMENTATION_files src/OSUtilsDummy.cpp)

set (_FRAMEWORK_PLATFORM_SPECIFIC_DUMMY

${ACCELEROMETER_IMPLEMENTATION_files}
${ORIENTATION_IMPLEMENTATION_files}
${AMERA_IMPLEMENTATION_files}
${DYNLOAD_IMPLEMENTATION_files}
${OSUTILS_IMPLEMENTATION_files}
)


ADD_IMPLEMENTATIONS(_FRAMEWORK_PLATFORM_SPECIFIC_DUMMY "dummy" _dummy_implementations)
ADD_CURRENT_LIST_DIR(SRC_FRAMEWORK_PLATFORM_SPECIFIC_DUMMY ${_FRAMEWORK_PLATFORM_SPECIFIC_DUMMY})
set(SRC_FRAMEWORK_PLATFORM_SPECIFIC_DUMMY ${SRC_FRAMEWORK_PLATFORM_SPECIFIC_DUMMY} CMakeLists.txt)

source_group("src\\framework\\platform_specific\\dummy\\include" FILES ${SRC_FRAMEWORK_PLATFORM_SPECIFIC_DUMMY_INCLUDE})
source_group("src\\framework\\platform_specific\\dummy\\src" FILES ${SRC_FRAMEWORK_PLATFORM_SPECIFIC_DUMMY})
set_source_files_properties(CMakeLists.txt PROPERTIES HEADER_FILE_ONLY TRUE)

endif(NOT ANDROID AND NOT IOS )
